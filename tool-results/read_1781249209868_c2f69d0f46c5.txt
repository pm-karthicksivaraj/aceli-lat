    23→function calcTitleLayout(title, maxWidthTwips, preferredPt = 40, minPt = 24) {
    24→  const charWidth = (pt) => pt * 20;
    25→  const charsPerLine = (pt) => Math.floor(maxWidthTwips / charWidth(pt));
    26→  let titlePt = preferredPt;
    27→  let lines;
    28→  while (titlePt >= minPt) {
    29→    const cpl = charsPerLine(titlePt);
    30→    if (cpl < 2) { titlePt -= 2; continue; }
    31→    lines = splitTitleLines(title, cpl);
    32→    if (lines.length <= 3) break;
    33→    titlePt -= 2;
    34→  }
    35→  if (!lines || lines.length > 3) {
    36→    const cpl = charsPerLine(minPt);
    37→    lines = splitTitleLines(title, cpl);
    38→    titlePt = minPt;
    39→  }
    40→  return { titlePt, titleLines: lines };
    41→}
    42→
    43→function splitTitleLines(title, charsPerLine) {
    44→  if (title.length <= charsPerLine) return [title];
    45→  const breakAfter = new Set([..."\u2018\u2019\u201C\u201D,\u3002\u3001\uFF1B\uFF1A\uFF01\uFF1F\u7684\u4E0E\u548C\u53CA\u4E4B\u5728\u4E8E\u4E3A-_-\u2014\u2013\u00B7/ \t"]);
    46→  const lines = [];
    47→  let remaining = title;
    48→  while (remaining.length > charsPerLine) {
    49→    let breakAt = -1;
    50→    for (let i = charsPerLine; i >= Math.floor(charsPerLine * 0.6); i--) {
    51→      if (i < remaining.length && breakAfter.has(remaining[i - 1])) { breakAt = i; break; }
    52→    }
    53→    if (breakAt === -1) {
    54→      const limit = Math.min(remaining.length, Math.ceil(charsPerLine * 1.3));
    55→      for (let i = charsPerLine + 1; i < limit; i++) {
    56→        if (breakAfter.has(remaining[i - 1])) { breakAt = i; break; }
    57→      }
    58→    }
    59→    if (breakAt === -1) breakAt = charsPerLine;
    60→    lines.push(remaining.slice(0, breakAt).trim());
    61→    remaining = remaining.slice(breakAt).trim();
    62→  }
    63→  if (remaining) lines.push(remaining);
    64→  if (lines.length > 1 && lines[lines.length - 1].length <= 2) {
    65→    const last = lines.pop();
    66→    lines[lines.length - 1] += last;
    67→  }
    68→  return lines;
    69→}

    71→function calcCoverSpacing(params) {
    72→  const { titleLineCount = 1, titlePt = 36, hasSubtitle = false, hasEnglishLabel = false, metaLineCount = 0, fixedHeight = 800, pageHeight = 16838, marginTop = 0, marginBottom = 0 } = params;
    73→  const SAFETY = 1200;
    74→  const usableHeight = pageHeight - marginTop - marginBottom - SAFETY;
    75→  const titleHeight = titleLineCount * (titlePt * 23 + 200);
    76→  const subtitleHeight = hasSubtitle ? (12 * 23 + 600) : 0;
    77→  const englishLabelHeight = hasEnglishLabel ? (9 * 23 + 600) : 0;
    78→  const metaHeight = metaLineCount * (10 * 23 + 100);
    79→  const implicitParaHeight = 3 * 300;
    80→  const contentHeight = titleHeight + subtitleHeight + englishLabelHeight + metaHeight + fixedHeight + implicitParaHeight;
    81→  const remainingSpace = usableHeight - contentHeight;
    82→  const safeRemaining = Math.max(remainingSpace, 400);
    83→  const FOOTER_MIN = 800;
    84→  const rawTop = Math.floor(safeRemaining * 0.45);
    85→  const rawBottom = Math.floor(safeRemaining * 0.45);
    86→  const bottomSpacing = Math.max(rawBottom, FOOTER_MIN);
    87→  const topSpacing = Math.max(rawTop - Math.max(0, FOOTER_MIN - rawBottom), 400);
    88→  const midSpacing = Math.max(safeRemaining - topSpacing - bottomSpacing, 0);
    89→  return { topSpacing, midSpacing, bottomSpacing };
    90→}

   134→  children.push(new Paragraph({
   135→    indent: { left: padL, right: padR },
   136→    border: { top: { style: BorderStyle.SINGLE, size: 2, color: CP.accent, space: 8 } },
   137→    spacing: { before: 200 },
   138→    children: [
   139→      new TextRun({ text: config.footerLeft || "", size: 16, color: CP.footerColor, font: { ascii: "Arial" } }),
   140→      new TextRun({ text: "                                        " }),
   141→      new TextRun({ text: config.footerRight || "", size: 16, color: CP.footerColor, font: { ascii: "Arial" } }),
   142→    ],
   143→  }));
   144→  return [new Table({
   145→    width: { size: 100, type: WidthType.PERCENTAGE },
   146→    borders: allNoBorders,
   147→    rows: [new TableRow({
   148→      height: { value: 16838, rule: "exact" },
   149→      children: [new TableCell({ shading: { type: ShadingType.CLEAR, fill: CP.bg }, borders: noBorders, children })],
   150→    })],
